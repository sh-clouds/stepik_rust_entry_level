use std::io;
use std::str::FromStr;

fn input<T: FromStr>() -> T 
where
    T::Err: std::fmt::Debug,
{
    let mut input = String::new();
    std::io::stdin()
        .read_line(&mut input)
        .expect("Failed to read input");
    input.trim().parse::<T>().expect("Failed to parse input")
}

fn main() {
    let hours: u8 = input();
    twelve_h_clck(hours);   
}


fn twelve_h_clck(hours: u8) {
    if hours == 0
    {
        println!("12 midnight");
    }
    else if hours > 0 && hours < 12
    {
        println!("{hours} am");
    }
    else if hours == 12
    {
        println!("12 noon");
    }    
    else
    {
        println!("{} pm", hours-12);
    }       
}
