Наибольшее кратное

В редакторе кода представлена неполная программа, которую необходимо доделать. Программа должна считать два целых числа a (i32) и b (u16), а затем вывести следующее наибольшее целое число к a, которое кратно числу b.

Нахождение числа реализуйте в функции lsg_multi() !

32
2

fn lsg_multi(a: i32, b: u16) -> i32{
    let mut c: i32 = a + 1; 
    while c % (b as i32) != 0 {
        c += 1
    }
    return c;
}

fn main() {
    let mut input_a = String::new();
    let mut input_b = String::new();
    
    std::io::stdin().read_line(&mut input_a).expect("Failed to read line");
    std::io::stdin().read_line(&mut input_b).expect("Failed to read line");
    
    let a: i32 = input_a.trim().parse().expect("Please enter a valid number");
    let b: u16 = input_b.trim().parse().expect("Please enter a valid number");
    
    println!("{}", lsg_multi(a, b));
}
