Считывание реализуйте в функции get_input() с возвратом считанного значения!

fn main() {
    let mut str = get_input();
    let mut n : i32;
    let mut f : bool = true;
    //print!("{}", str);     
    n = str.trim().parse::<i32>().expect("Failed to parse input");
    //print!("{}", n);  
        
    for i in 2..n
    {
        if n % i == 0
        {
            if f == true
            {
                f = false;
            }
            else
            {
                print!(" ");
            }            
            print!("{i}");
        }    
    }
}

fn get_input() -> String {
    let mut str = String::new();
    std::io::stdin()
        .read_line(&mut str)
        .expect("failed to read stdin");
    str
}
