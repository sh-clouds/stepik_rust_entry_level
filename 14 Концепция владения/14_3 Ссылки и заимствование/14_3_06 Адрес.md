fn get_input() -> String {
    let mut str = String::new();
    std::io::stdin()
        .read_line(&mut str)
        .expect("failed to read stdin");
    str.trim().to_string()
}

fn main() {
    let mut str = get_input();
    print!("{:p}", &str);
