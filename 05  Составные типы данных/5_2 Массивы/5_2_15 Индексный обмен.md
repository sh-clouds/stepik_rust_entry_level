use std::io;
use std::str::FromStr;
use std::fmt::Debug;
fn read_number<T: FromStr>() -> T where <T as FromStr>::Err: Debug {
    let mut user_input = String::new();
    io::stdin().read_line(&mut user_input).expect("Не удалось прочитать ввод");
    user_input.trim().parse().expect("Не удалось преобразовать в число")
}

fn main() {
    let mut arr = [-621.5, 11.1, 2.0, -7.123, 0.125, 0.0, 0.000051789];
    let i1: usize = read_number();   
    let i2: usize = read_number();
    let mut temp: f64;
    temp = arr[i1];
    arr[i1] = arr[i2];
    arr[i2] = temp;    
    println!("{:.9?}", arr);  
}
