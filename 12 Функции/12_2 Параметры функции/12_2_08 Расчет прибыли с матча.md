use std::io;
use std::str::FromStr;

fn input1<T: FromStr>() -> T 
where
    T::Err: std::fmt::Debug,
{
    let mut input1 = String::new();
    std::io::stdin()
        .read_line(&mut input1)
        .expect("Failed to read input");
    input1.trim().parse::<T>().expect("Failed to parse input")
}

fn input() {    
    // A = one_tkt_cst[0], B = one_tkt_cst[1], C = one_tkt_cst[2]
    let mut one_tkt_cst = [0u32; 3];
    for i in 0..3 
    {
        one_tkt_cst[i] = input1::<u32>();
    }

    // A = num_tks_sld[0], B = num_tks_sld[1], C = num_tks_sld[2]
    let mut num_tks_sld = [0u16; 3];
    for i in 0..3

}

    input();
}

fn calc_profit(one_tkt_cst: [u32; 3], num_tks_sld: [u16; 3]) {
    let mut profit: u32 = 0;
    for i in 0..3 
    {
        profit += one_tkt_cst[i] * num_tks_sld[i] as u32;

}
