5
10

Из 5.000 и 10.000 больше 10.000

use std::io;
use std::str::FromStr;

fn input<T: FromStr>() -> T 
where
    T::Err: std::fmt::Debug,
{
    let mut input = String::new();
    std::io::stdin()
        .read_line(&mut input)
        .expect("Failed to read input");
    input.trim().parse::<T>().expect("Failed to parse input")
}

fn main() {
    let f1: f64 = input(); 
    let f2: f64 = input(); 
    let mut f3: f64;    
    if f1 > f2 {f3 = f1;} else {f3 = f2;}
    println!("Из {:.3} и {:.3} больше {:.3}", f1, f2, f3);          
}
