В редакторе кода представлен массив чисел. Дополните код и считайте введенный индекс.

Если считанное число меньше 0, то выведите сообщение: Отрицательный индекс приводит к панике.
Если больше максимального индекса массива, то выведите сообщение: Попытка выхода за пределы массива.
Если же индекс корректен, вывести (до 3 знаков) по этому индексу соответствующий элемент с сообщением: Элемент с индексом {index} равен {item}.

-1

Отрицательный индекс приводит к панике

Sample Input 2:
0

Элемент с индексом 0 равен -2.500

use std::io;
use std::str::FromStr;

fn input<T: FromStr>() -> T 
where
    T::Err: std::fmt::Debug,
{
    let mut input = String::new();
    std::io::stdin()
        .read_line(&mut input)
        .expect("Failed to read input");
    input.trim().parse::<T>().expect("Failed to parse input")
}

fn main() {
    let arr = [-2.5, 4.2, 9.1, 22.5, 30.0, 1445.123, 1000000.0, 0.001, 0.5, -0.127];    
    let i1: i32 = input(); 

    if i1 < 0
    {
        println!("Отрицательный индекс приводит к панике");
    }
    else
    {
        if i1 as usize > (arr.len()-1)
        {
            println!("Попытка выхода за пределы массива");
        } 
        else
        {
            if i1 > 0
            {
                println!("Элемент с индексом {} равен {:.3}", i1, arr[i1 as usize]);
            }
            else
            {
                println!("Элемент с индексом 0 равен {:.3}", arr[i1 as usize]);        
            }                   
        }
    }      
}
