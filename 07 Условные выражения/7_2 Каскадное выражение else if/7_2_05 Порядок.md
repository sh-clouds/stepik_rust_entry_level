Напишите программу, которая считывает три вещественных числа и выводит (до 1 знака) их в порядке возрастания. Гарантируется, что одинаковых чисел нет!

3
2
1

15.5
14.2
13.9

13.9, 14.2, 15.5

use std::io;
use std::str::FromStr;

fn input<T: FromStr>() -> T 
where
    T::Err: std::fmt::Debug,
{
    let mut input = String::new();
    std::io::stdin()
        .read_line(&mut input)
        .expect("Failed to read input");
    input.trim().parse::<T>().expect("Failed to parse input")
}
