 Так, если a = 5, то вывод будет выглядеть следующим образом:

 5: 00000000000000000000000000000101 :a

-6: 11111111111111111111111111111010 :a
 5: 00000000000000000000000000000101 :!a

0

 0: 00000000000000000000000000000000 :a

use std::io;
use std::str::FromStr;

fn input<T: FromStr>() -> T 
where
    T::Err: std::fmt::Debug,
{
    let mut input = String::new();
    std::io::stdin()
        .read_line(&mut input)
        .expect("Failed to read input");
    input.trim().parse::<T>().expect("Failed to parse input")
}

fn main() {       
    let mut a: i32 = input();
    
    if a>=0 
    { 
        println!(" {}: {:032b} :a", a, a); 
    }
    else
    {
        println!("{}: {:032b} :a", a, a); 
    }
    
    if !a>=0 
    {    
        println!(" {}: {:032b} :!a", !a, !a); 
    }
    else
    {    
        println!("{}: {:032b} :!a", !a, !a); 
    }    
    
}
