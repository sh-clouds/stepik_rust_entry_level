use std::io;
use std::str::FromStr;

fn input<T: FromStr>() -> T 
where
    T::Err: std::fmt::Debug,
{
    let mut input = String::new();
    std::io::stdin()
        .read_line(&mut input)
        .expect("Failed to read input");
    input.trim().parse::<T>().expect("Failed to parse input")
}

fn main() {       
    let mut x: i8 = input();  
    let mut n: u8 = input();   
    //x = 3;
    //n = 1;

    println!("x до очистки от 7 до {}-го бита", n); 
    println!("в двоичной записи: {:08b}", x);     
    println!("в десятичной записи: {}\n", x); 

    println!("x после очистки от 7 до {}-го бита", n); 
    println!("в двоичной записи: {:08b}", x & ((1<<n)-1));     
    println!("в десятичной записи: {}\n", x & ((1<<n)-1));        
}
