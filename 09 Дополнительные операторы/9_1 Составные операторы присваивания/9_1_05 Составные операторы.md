use std::io;
use std::str::FromStr;

fn input<T: FromStr>() -> T 
where
    T::Err: std::fmt::Debug,
{
    let mut input = String::new();
    std::io::stdin()
        .read_line(&mut input)
        .expect("Failed to read input");
    input.trim().parse::<T>().expect("Failed to parse input")
}

fn main() {    
    let mut array = [1, 2, 3, 4, 5];    
    let mut x: i32 = input();
 
    array[0] += x;
    array[1] -= x;   
    array[2] *= x;  
    array[3] /= x;  
    array[4] %= x;      
    
    println!("{array:?}");     
}
