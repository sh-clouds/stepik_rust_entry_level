Напишите программу, которая считывает вещественное число и выводит его целое и дробную часть (до 3 знаков). Например, если введено число 123.456789, программа должна вывести:

123
0.457

fn read_float() -> f32 {
    let mut strnum = String::new();
    std::io::stdin().read_line(&mut strnum).expect("Could not read the input.");
    let num = strnum.trim().parse().expect("Incorrect input. Please enter the number");
    return num;
}

fn main() {
    let fnum = read_float();
    let fnum_as_i32 = fnum as i32;
    println!("{}\n{:.3}", fnum_as_i32, fnum - fnum_as_i32 as f32);
}
