Вы оказались в странном мире, где правила математики работают немного иначе. Здесь банкоматы сломались, и теперь они пополняют карты только купюрами в виде целых чисел, в то время как баланс вашей карты хранится в виде вещественного числа.

fn read_int() -> i32 {
    let mut strnum = String::new();
    std::io::stdin().read_line(&mut strnum).expect("Could not read the input.");
    let num = strnum.trim().parse().expect("Incorrect input. Please enter the number");
    return num;
}

fn read_float() -> f32 {
    let mut strnum = String::new();
    std::io::stdin().read_line(&mut strnum).expect("Could not read the input.");
    let num = strnum.trim().parse().expect("Incorrect input. Please enter the number");
    return num;
}

    let num = read_float();
