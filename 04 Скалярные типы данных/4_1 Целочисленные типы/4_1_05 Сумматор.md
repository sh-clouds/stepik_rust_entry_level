use std::io;

fn main() {
    // put your Rust code here
    let mut s1 = String::new();
    let mut s2 = String::new();    
    
    io::stdin()
        .read_line(&mut s1) 
        .expect("Не удалось прочитать ввод"); 

    io::stdin()
        .read_line(&mut s2) 
        .expect("Не удалось прочитать ввод");   
    
    let num1: i32;
    let num2: i32;    
    num1 = s1
        .trim()
        .parse() 
        .expect("Пожалуйста, введите число!");
    
    num2 = s2
        .trim()
        .parse() 
        .expect("Пожалуйста, введите число!");    
  
    println!("{}", num1 + num2);
}
