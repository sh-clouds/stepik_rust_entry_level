Треугольник Паскаля

Напишите программу, которая считывает натуральное число n (u16) и выводит первые n строк треугольника Паскаля.

1

use std::io;
use std::str::FromStr;

fn input<T: FromStr>() -> T 
where
    T::Err: std::fmt::Debug,
{
    let mut input = String::new();
    std::io::stdin()
        .read_line(&mut input)
        .expect("Failed to read input");
    input.trim().parse::<T>().expect("Failed to parse input")
}

fn main() {
    let rows = input::<u16>();
    
    for i in 0..rows {
        let mut current_item = 1_u16;
        
        for _ in 0..(rows - 1 - i) {
            print!(" ");
        }
        
        print!("{current_item}");
        for k in 1..=i {
            current_item = current_item * (i + 1 - k) / k;
            print!(" {}", current_item); 
        }
        println!("");
    }
}
