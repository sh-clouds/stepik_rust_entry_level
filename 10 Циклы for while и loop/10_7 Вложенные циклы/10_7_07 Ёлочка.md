use std::io;
use std::str::FromStr;

fn input<T: FromStr>() -> T 
where
    T::Err: std::fmt::Debug,
{
    let mut input = String::new();
    std::io::stdin()
        .read_line(&mut input)
        .expect("Failed to read input");
    input.trim().parse::<T>().expect("Failed to parse input")
}

fn main() 
{     
    let mut n: i32 = input(); 
    let mut k: i32;
    //n = 6;
    
    for i in 1..=n  
    {      
        k = n - i;
        for j in 0..k
        {
            print!(" ");  
        }
        
        k = (i-1)*2 + 1;
        for j in 0..k
        { 
            print!("*"); 
        }
        println!("");
    }
}
