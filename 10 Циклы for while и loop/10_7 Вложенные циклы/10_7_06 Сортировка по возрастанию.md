use std::io;
use std::str::FromStr;

fn input<T: FromStr>() -> T 
where
    T::Err: std::fmt::Debug,
{
    let mut input = String::new();
    std::io::stdin()
        .read_line(&mut input)
        .expect("Failed to read input");
    input.trim().parse::<T>().expect("Failed to parse input")
}

fn main() 
{
    let mut a = [10000_i32; 10];
    let mut b = [10000_i32; 12]; //temp       
    let mut d: i32;
    let mut f: bool = false;
    let mut p: usize = 0; //pointer b

    for _ in 0..10  
    {       
        for i in 0..10
        {
            a[i] = b[i];
        }
        
        d =  input();
        p = 0;
        f = false;
        
        for i in 0..10
        {
            if (d < a[i]) && (f == false) 
            {
                f = true;
                b[p] = d;                
                p += 1;
                b[p] = a[i];
            }
            else
            {
                b[p] = a[i];                
            }
            p += 1;             
        }
    } 
    for i in 0..10
    {
        a[i] = b[i];
    }
    println!("{:?}", a);      
}
