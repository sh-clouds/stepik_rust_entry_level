use std::io;
use std::str::FromStr;

fn input<T: FromStr>() -> T 
where
    T::Err: std::fmt::Debug,
{
    let mut input = String::new();
    std::io::stdin()
        .read_line(&mut input)
        .expect("Failed to read input");
    input.trim().parse::<T>().expect("Failed to parse input")
}

fn main() {            
    let mut start: i8 = input();  
    let mut end: i8 = input(); 
    
    for i in start..end {
        println!("{i} * 2 = {}", i*2);
    }      
}
