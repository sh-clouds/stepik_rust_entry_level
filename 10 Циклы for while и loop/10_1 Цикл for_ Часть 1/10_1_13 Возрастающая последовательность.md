use std::io;
use std::str::FromStr;

fn input<T: FromStr>() -> T 
where
    T::Err: std::fmt::Debug,
{
    let mut input = String::new();
    std::io::stdin()
        .read_line(&mut input)
        .expect("Failed to read input");
    input.trim().parse::<T>().expect("Failed to parse input")
}

fn main() {            
    let mut a: i32 = input();  
    let mut b: i32 = input(); 
    let mut temp: i32 = 0;

    if a > b
    {
        temp = a;
        a = b;
        b = temp;
    }
    
    for i in a..=b {
       println!("{i}"); 
    }    
}
