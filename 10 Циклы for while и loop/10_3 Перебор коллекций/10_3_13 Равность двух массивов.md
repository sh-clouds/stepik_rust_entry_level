Равность двух массивов

Иначе вывести Массивы {:?} и {:?} не равны.

use std::io;
use std::str::FromStr;

fn input<T: FromStr>() -> T 
where
    T::Err: std::fmt::Debug,
{
    let mut input = String::new();
    std::io::stdin()
        .read_line(&mut input)
        .expect("Failed to read input");
    input.trim().parse::<T>().expect("Failed to parse input")
}

fn main() {
    let mut a1 = [0; 10];
    let mut a2 = [0; 10];  
    let mut f: bool = true;
 
    for i in 0..10 { 
        a1[i] = input();    
    }
    for i in 0..10 { 
        a2[i] = input();    
    }    
    
    for i in 0..10 { 
        if a1[i] != a2[i]
        {
           f = false; 
        }
    }      
    
    if f == true
    {
        println!("Массивы {:?} и {:?} равны", a1, a2);
    }
    else
    {
        println!("Массивы {:?} и {:?} не равны", a1, a2);
    }      
}
