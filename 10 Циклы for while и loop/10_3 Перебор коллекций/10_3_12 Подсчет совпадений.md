Напишите программу, которая считывает целое число num, натуральное значение n (u8) и столько же вводимых пользователем чисел, а затем выводит количество элементов больших num для считанной последовательности в виде сообщения: Количество элементов больших {num}: {}.

Тестовые данные ✅

Sample Input:
2
5
1
2
3
4
5

Sample Output:
Количество элементов больших 2: 3

use std::io;
use std::str::FromStr;

fn input<T: FromStr>() -> T 
where
    T::Err: std::fmt::Debug,
{
    let mut input = String::new();
    std::io::stdin()
        .read_line(&mut input)
        .expect("Failed to read input");
    input.trim().parse::<T>().expect("Failed to parse input")
}

fn main() {
    
    let mut num: i32 = input(); 
    let mut n: u8 = input();
    let mut val: i32;
    let mut i1: i32 = 0;

    for i in 0..n { 
        val = input();
        if val > num 
        {
            i1 += 1;
        }      
    }
    print!("Количество элементов больших {num}: {i1}");
}
