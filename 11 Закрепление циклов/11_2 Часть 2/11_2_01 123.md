fn main() {
    let arr: [u16; 3] = [1, 2, 3];
    // xyz - трехзначное число
    for &x in &arr {
        for &y in &arr {
            for &z in &arr {                
                if x != y && x != z && y != z {
                    let num = x * 100 + y * 10 + z;
                    println!("{}", num);
                }
            }
        }
    }
}
