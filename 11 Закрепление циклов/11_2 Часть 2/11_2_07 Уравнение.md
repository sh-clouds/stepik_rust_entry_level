Уравнение

Напишите программу, которая выводит целочисленные решения, если таковые имеются для уравнения:

4x−2y+3z=20, xyz<15  и  x^2 + y^2 +z^2 > 

где x - целое число в пределах [−10, 10], y находится в пределах [−3, 3] и z находится в пределах [2, 6].

Вывод оформить в виде:

 x  y  z
 0  0  0
-1 -1 -1
.........
Напишите программу. Тестируется через stdin → stdout

fn main() {
    println!(" x  y  z");    
    for x in -10..=10 
    {
        for y in -3..=3 
        {
            for z in 2..=6 
            {
                if x*y*z < 15
                {
                    if 4*x - 2*y + 3*z ==20
                    {
                        if x*x + y*y + z*z > 8
                        {
                            if x == 10
                            {
                                print!("10");
                            }
                            else if x < 0
                            {
                                print!("{x}");
                            }
                            else
                            {
                                print!(" {x}"); 
                            }
                            
                            if y < 0
                            {
                                print!(" {y}");
                            }
                            else
                            {
                                print!("  {y}"); 
                            }

                            println!("  {z}"); 
                        } 
                    }                                
                }                
            }  
        }        
    }
}
