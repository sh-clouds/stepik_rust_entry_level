Напишите программу, которая считывает натуральное число n (u32) и вещественное значение x, а затем выводит (до 5 знаков) значение P вычисленное по формуле:

P=∏_i=0_n (x+1)/(x^2+4)
  
Тестовые данные ✅

Sample Input:
1
1

Sample Output:
0.19753

Напишите программу. Тестируется через stdin → stdout

use std::io;
use std::str::FromStr;

fn input<T: FromStr>() -> T 
where
    T::Err: std::fmt::Debug,
{
    let mut input = String::new();
    std::io::stdin()
        .read_line(&mut input)
        .expect("Failed to read input");
    input.trim().parse::<T>().expect("Failed to parse input")
}

fn main() {
    let n: i32 = input();
    let x: f64 = input();
    let mut m: f64 = 1.0;
    
    for _ in 0..=n 
    {    
        m *= (x + 1.0) / ((x * x) /2.0  + 4.0); 
    }  
    
    println!("{m:.5}");
}
