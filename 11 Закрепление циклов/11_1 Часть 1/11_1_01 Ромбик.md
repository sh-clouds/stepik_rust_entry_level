fn print_rhombus(size: i32) {
    for row in 0..size {
        println!(
            "{}{}",
            " ".repeat((size - row - 1) as usize),
            "*".repeat((2 * row + 1) as usize)
        );
    }
    for row in (0..size - 1).rev() {
        println!(
            "{}{}",
            " ".repeat((size - row - 1) as usize),
            "*".repeat((2 * row + 1) as usize)
        );
    }
}

use std::io;

fn main() {
    let mut input_string = String::new();
    io::stdin().read_line(&mut input_string).unwrap();
    let size: i32 = input_string.trim().parse().unwrap();
    print_rhombus(size);
}
