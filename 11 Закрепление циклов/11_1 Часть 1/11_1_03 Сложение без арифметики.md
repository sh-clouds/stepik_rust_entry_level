fn sum_bitwise(mut a: i32, mut b: i32) -> i32 {
    //Computes the sum of two integers a and b using only bitwise operations.
    while b != 0 {
        let carry = a & b; // Determine carry
        a = a ^ b; // Sum without carry
        b = carry << 1; // Carry shifted left by 1 bit
    }
    a
}

use std::io;

fn main() {
    let mut input1 = String::new();
    io::stdin().read_line(&mut input1).unwrap();
    let num1: i32 = input1.trim().parse().unwrap();

    let mut input2 = String::new();
    io::stdin().read_line(&mut input2).unwrap();
    let num2: i32 = input2.trim().parse().unwrap();

    let result = sum_bitwise(num1, num2);

}
