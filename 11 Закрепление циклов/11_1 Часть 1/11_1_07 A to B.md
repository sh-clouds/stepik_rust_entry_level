use std::io::stdin;
use std::str::FromStr;

fn input<T: FromStr>() -> Result<T, T::Err> {
    let mut s = String::new();
    stdin().read_line(&mut s).unwrap();
    s.trim().parse()
}

fn main() {
    let a: u8 = input().unwrap();
    let b: u8 = input().unwrap();
    println!("{}", (a ^ b).count_ones());
}
